% Slide 25: World Cup Simulator - Mathematical Formulation
\begin{frame}{World Cup Simulator: Mathematical Formulation}
  Match goals are modeled as independent Poisson distributions:
  \[ P(X = k) = \frac{\lambda^k e^{-\lambda}}{k!} \]
  Where:
  \begin{itemize}
    \item $P(X = k)$ is the probability of a team scoring $k$ goals.
    \item $\lambda$ represents the expected goals calculated from:
      \[ \lambda_{\text{Team A}} = \text{Attack Strength}_{\text{Team A}} \times \text{Defense Vulnerability}_{\text{Team B}} \times \text{Average Goals} \]
  \end{itemize}
  Draw probabilities are resolved by computing the cross-product of goal distributions.
\end{frame}

% Slide 26: World Cup Simulator - Elo Rating Engine
\begin{frame}{World Cup Simulator: Elo Rating Engine}
  Elo ratings are dynamically calculated based on historical match outcomes:
  \[ R_{\text{new}} = R_{\text{old}} + K(W - W_e) \]
  Where:
  \begin{itemize}
    \item $K$ is the match importance factor (higher multiplier for World Cup matches).
    \item $W$ is the actual result (1 for win, 0.5 for draw, 0 for loss).
    \item $W_e$ is the expected win probability based on rating difference:
      \[ W_e = \frac{1}{10^{-(R_A - R_B)/400} + 1} \]
  \end{itemize}
\end{frame}

% Slide 27: World Cup Simulator - Group & Knockout Logic
\begin{frame}{World Cup Simulator: Group \& Knockout Logic}
  \begin{itemize}
    \item \textbf{48-Team Expansion Complexity:}
      \begin{itemize}
        \item 12 groups of 4 teams.
        \item Top 2 from each group advance.
        \item The 8 best third-place teams advance to fill the Round of 32.
      \end{itemize}
    \item \textbf{Algorithm Design:}
      \begin{itemize}
        \item Simulates all group games, resolves rankings using tie-breakers (points, goal difference, goals scored).
        \item Collects all third-place teams, sorts them, and routes them to correct knockout branches.
        \item Knockout matches use extra time and penalty shootouts if a draw is simulated.
      \end{itemize}
  \end{itemize}
\end{frame}

% Slide 36: System Design - PostgreSQL & PostGIS Spatial Indexes
\begin{frame}{System Design: PostgreSQL \& PostGIS Spatial Indexes}
  \begin{itemize}
    \item \textbf{Spatial Data Type:} Store coordinates as `GEOGRAPHY(Point, 4326)` instead of separate float columns.
    \item \textbf{Spatial Indexing:} Create a GIST index on coordinates:
      \[ \text{CREATE INDEX idx\_photos\_geom ON photos USING GIST(geom);} \]
    \item \textbf{Why GIST:} Standard B-Tree indexes cannot handle multi-dimensional queries. GIST indexes group spatial boundaries into hierarchical structures, speeding up regional searches.
  \end{itemize}
\end{frame}

% Slide 37: System Design - Viewport Bounding Box Queries
\begin{frame}[fragile]{System Design: Viewport Bounding Box Queries}
  \begin{itemize}
    \item \textbf{Concept:} The client only requests markers that fit inside the current visible map boundary.
    \item \textbf{Implementation:} Map emits `moveend` event $\to$ gets bounding box coordinates $\to$ requests API data.
  \end{itemize}
\begin{lstlisting}[language=SQL]
SELECT id, filename, lat, lng, ai_description 
FROM photos 
WHERE geom && ST_MakeEnvelope(west, south, east, north, 4326);
\end{lstlisting}
  \begin{itemize}
    \item \textbf{Benefit:} Reduces API query size from 10k+ rows to just a few dozen, keeping client maps responsive.
  \end{itemize}
\end{frame}

% Slide 38: System Design - S3 Object Storage & CDN
\begin{frame}{System Design: S3 Object Storage \& CDN}
  \begin{itemize}
    \item \textbf{The Bottleneck:} Serving image assets directly through the Express API server blocks request loops and limits server CPU resources.
    \item \textbf{Production Storage Architecture:}
  \end{itemize}
  \begin{center}
    Client Upload $\to$ API Presigned URL $\to$ Direct Upload to AWS S3 / Cloudflare R2
  \end{center}
  \begin{itemize}
    \item \textbf{Delivery:} Front the storage bucket with a CDN (Cloudflare or CloudFront) to cache images closer to users and offload server traffic.
  \end{itemize}
\end{frame}

% Slide 39: System Design - Secure Cookie-Based Auth Rotation
\begin{frame}{System Design: Secure Cookie-Based Auth Rotation}
  \begin{itemize}
    \item \textbf{Current Risk:} Storing JWTs in `LocalStorage` makes them vulnerable to Cross-Site Scripting (XSS) attacks.
    \item \textbf{Production Auth Design:}
      \begin{itemize}
        \item Store short-lived access tokens in memory or secure HttpOnly cookies.
        \item Store long-lived refresh tokens in HttpOnly, Secure, SameSite cookies.
        \item Implement refresh token rotation on the backend to detect token theft.
      \end{itemize}
  \item \textbf{Benefit:} Protects user sessions from client-side script inspection.
  \end{itemize}
\end{frame}

% Slide 40: GeoPhoto - Practical Engineering Challenges
\begin{frame}{GeoPhoto: Practical Engineering Challenges}
  \begin{itemize}
    \item \textbf{GPS Metadata Reliability:} 
      \begin{itemize}
        \item Over $50\%$ of user photos had GPS metadata stripped (due to social media exports/privacy settings).
        \item \textbf{Solution:} Designed an interactive map click coordinate listener to ensure seamless user fallback.
      \end{itemize}
    \item \textbf{Asynchronous UI State Lag:}
      \begin{itemize}
        \item Since AI generation runs in the background, updating the UI immediately after upload required state pooling or manual regeneration triggers.
      \end{itemize}
    \item \textbf{API Quota & Rate Limits:}
      \begin{itemize}
        \item Rapid uploads during testing triggered Gemini 429 rate limit errors.
        \item \textbf{Solution:} Added exponential backoff retry algorithms to handle network congestion gracefully.
      \end{itemize}
  \end{itemize}
\end{frame}

% Slide 41: World Cup Simulator - Constraints & Edge Cases
\begin{frame}{World Cup Simulator: Constraints \& Edge Cases}
  \begin{itemize}
    \item \textbf{Computational Bottlenecks:}
      \begin{itemize}
        \item Evaluating complex third-place ranking comparisons for 10,000 runs inside standard loops caused massive lags.
        \item \textbf{Solution:} Vectorized calculations using NumPy matrices.
      \end{itemize}
    \item \textbf{Predictive Modeling Drift:}
      \begin{itemize}
        \item Static Elo ratings struggle to capture sudden squad alterations, injuries, or tactical shifts.
      \item Historical data skew makes predicting outliers or extreme dark horses difficult.
      \end{itemize}
    \item \textbf{Tie-breaker edge cases:}
      \begin{itemize}
        \item Complex tie-breaker sequences (fair play points, drawing of lots) are hard to model mathematically.
      \end{itemize}
  \end{itemize}
\end{frame}

% Slide 42: What Didn't Work Perfectly & Key Learnings
\begin{frame}{What Didn't Work Perfectly \& Key Learnings}
  \begin{itemize}
    \item \textbf{Local File Watcher Limits:}
      \begin{itemize}
        \item The frontend live watch system (`npm run dev`) failed on macOS due to OS watch handles reaching maximum capacity.
        \item \textbf{Learning:} Realized the importance of identifying and documenting alternative commands (`npm start` / watcher adjustments) in developer guides.
      \end{itemize}
    \item \textbf{Testing Leaflet in JSDOM:}
      \begin{itemize}
        \item Leaflet map containers depend on actual browser layout dimensions (offsets/widths). Testing them in headless JSDOM environments was extremely difficult.
        \item \textbf{Learning:} Had to write elaborate mocks for Leaflet APIs, teaching me that full E2E testing (Playwright) is far better suited for maps than unit tests.
      \end{itemize}
  \end{itemize}
\end{frame}
