% Slide 7: Stage 1: Client EXIF Parsing
\begin{frame}{Stage 1: Client EXIF Ingestion}
  \begin{itemize}
    \item \textbf{Concept:} Image files contain metadata blocks storing GPS tags captured at the moment of exposure.
    \item \textbf{Client-First Ingestion:}
      \begin{itemize}
        \item UploadModal reads the raw file locally using \texttt{exifr}.
        \item Converts GPS coordinate formats to decimal values on the client.
        \item Updates latitude and longitude state hooks instantly.
      \end{itemize}
    \item \textbf{Value:} Backend remains fast as it doesn't parse image binary streams for coordinates.
  \end{itemize}
\end{frame}

% Slide 8: Stage 1: Advanced Telemetry (Altitude & Yaw/Pitch/Roll)
\begin{frame}{Stage 1: Advanced Camera Telemetry}
  \begin{itemize}
    \item \textbf{Context:} Knowing where a sensor is isn't enough; we need to know where the lens was pointing.
    \item \textbf{Extended Metadata Fields:}
      \begin{itemize}
        \item \textbf{Altitude:} Stored in EXIF tags, giving coordinate heights.
        \item \textbf{Yaw, Pitch, Roll:} Camera gimbal orientation stored in XMP schemas.
      \end{itemize}
    \item \textbf{Value:} Allows the system to project the exact ground footprint of the image onto the map instead of rendering a point.
  \end{itemize}
\end{frame}

% Slide 10: Stage 1: Fallback UI (Mini-Map clicker)
\begin{frame}{Stage 1: Manual Coordinate Overrides}
  \begin{itemize}
    \item \textbf{The Issue:} Over $50\%$ of web images have GPS metadata stripped due to privacy filters.
    \item \textbf{Fallback Ingestion Flow:}
      \begin{itemize}
        \item If EXIF fails, coordinate inputs are displayed.
        \item Users click on an interactive mini Leaflet map embedded in the modal.
        \item Captures Leaflet click events and updates coordinate inputs immediately.
      \end{itemize}
    \item \textbf{Benefit:} Ensures all images are successfully associated with coordinates.
  \end{itemize}
\end{frame}

% Slide 11: Stage 1: Viewport Performance (Marker Clustering)
\begin{frame}{Stage 1: Marker Clustering}
  \begin{itemize}
    \item \textbf{The Bottleneck:} Rendering hundreds of individual DOM markers crashes client-side mapping threads.
    \item \textbf{Clustering Strategy:}
      \begin{itemize}
        \item Wraps Leaflet markers in \texttt{react-leaflet-cluster} (Supercluster library).
        \item Aggregates close coordinates into single nodes with count badges.
        \item Zooming in expands nodes dynamically.
      \end{itemize}
    \item \textbf{Benefit:} Keeps map panning at 60 FPS, even over dense areas.
  \end{itemize}
\end{frame}

% Slide 12: Stage 1: Viewport Event Handlers
\begin{frame}{Stage 1: Visible Screen Bounding Boxes}
  \begin{itemize}
    \item \textbf{Concept:} The client map only requests images visible on the screen.
    \item \textbf{Workflow:}
      \begin{itemize}
        \item Map captures Leaflet's \texttt{moveend} events.
        \item Retrieves screen bounds: Southwest and Northeast decimal coordinates.
        \item Sends bounding box request: `GET /photos?bbox=south,west,north,east`.
      \end{itemize}
    \item \textbf{Optimization:} Viewport requests are debounced to prevent duplicate API hits.
  \end{itemize}
\end{frame}

% Slide 17: Stage 2: Image Processing (Sharp)
\begin{frame}{Stage 2: Image Normalization}
  \begin{itemize}
    \item \textbf{The Bug:} Images taken on mobile devices contain rotation EXIF tags that cause browsers to render them sideways.
    \item \textbf{Sharp Solution:}
      \begin{itemize}
        \item Upload router intercepts file buffers using `multer`.
        \item Passes buffers to `sharp`.
        \item Calls `.rotate()` to read EXIF orientation tags and permanently rotate pixels.
        \item Compresses file size to JPEG format at $80\%$ quality.
      \end{itemize}
    \item \textbf{Benefit:} Guarantees correct rendering across all browsers while saving $60\%$ disk space.
  \end{itemize}
\end{frame}

% Slide 20: Stage 3: AI Quota & Retry Logic
\begin{frame}{Stage 3: AI Quota & Retry Logic}
  \begin{itemize}
    \item \textbf{Challenge:} API rate limits (429) or remote drops (503) will occasionally crash uploads.
    \item \textbf{Retry Strategy (Exponential Backoff):}
      \begin{itemize}
        \item Catch API errors and pause.
        \item Retry up to 3 times.
        \item Delays double on each failure ($2\text{s} \to 4\text{s} \to 8\text{s}$).
      \end{itemize}
    \item \textbf{Failure Strategy:} Defaults to `NULL` in the database, leaving the core upload unaffected.
  \end{itemize}
\end{frame}

% Slide 23: Stage 4: Cryptographically Secure OTPs
\begin{frame}{Stage 4: Secure OTP Generation}
  \begin{itemize}
    \item \textbf{The Risk:} \texttt{Math.random()} is predictable. If a hacker reverse-engineers the seed, they can guess signup and reset codes.
    \item \textbf{Solution:} Replaced Math.random with Node's native \texttt{crypto.randomInt(10000000, 100000000)}.
    \item \textbf{Benefit:} Uses OS-level entropy to generate unpredictable 8-digit verification codes.
  \end{itemize}
\end{frame}

% Slide 24: Stage 4: Expiration & Single-Use revokes
\begin{frame}{Stage 4: Expiration \& Revocation}
  \begin{itemize}
    \item \textbf{Lifecycle Rules:}
      \begin{itemize}
        \item Codes are mapped in memory with an expiration: `expiresAt: Date.now() + 10 * 60 * 1000` (10-min TTL).
        \item Expired codes are automatically deleted.
        \item Valid codes are deleted instantly upon successful verification.
      \end{itemize}
    \item \textbf{Benefit:} Prevents brute-force attempts and replay attacks.
  \end{itemize}
\end{frame}

% Slide 33: Stage 6: PostGIS Migration
\begin{frame}{Stage 6: PostgreSQL & PostGIS}
  \begin{itemize}
    \item \textbf{SQLite Limits:} File locks on writes, lacks horizontal scale, and lacks spatial indexes.
    \item \textbf{PostGIS Database:}
      \begin{itemize}
        \item Store points inside a `geom GEOGRAPHY(Point, 4326)` column.
        \item Configure database triggers to synchronize `geom` values whenever lat/lng changes.
      \end{itemize}
  \end{itemize}

% Slide 34: Stage 6: GIST Spatial Indexing
\begin{frame}{Stage 6: GIST Spatial Indexing}
  \begin{itemize}
    \item \textbf{Concept:} B-Tree indexes cannot handle 2D coordinates. We use GIST (Generalized Search Tree) indexes.
    \item \textbf{Command:}
      \lstinline[language=SQL]{CREATE INDEX idx_photos_geom ON photos USING GIST(geom);}
    \item \textbf{Action:} Groups coordinate points into visible spatial grids. Bypasses full-table scans.
  \end{itemize}
\end{frame}

% Slide 35: Stage 6: Bounding Box Queries (ST_MakeEnvelope)
\begin{frame}[fragile]{Stage 6: Viewport Envelope Queries}
  \begin{itemize}
    \item \textbf{Algorithm:} Coordinates visible on screen are mapped to PostGIS envelopes.
  \end{itemize}
\begin{lstlisting}[language=SQL]
SELECT id, filename FROM photos 
WHERE geom && ST_MakeEnvelope(west, south, east, north, 4326);
\end{lstlisting}
  \begin{itemize}
    \item \textbf{Operator `\&\&`:} Bounding box intersection check. Bypasses expensive geometry calculations to return viewport data instantly.
  \end{itemize}
\end{frame}

% Slide 36: Stage 6: Proximity Queries on Linear Assets
\begin{frame}[fragile]{Stage 6: Proximity to Infrastructure}
  \begin{itemize}
    \item \textbf{Use Case:} Find all photos captured within 50 meters of a specific pipeline segment.
  \end{itemize}
\begin{lstlisting}[language=SQL]
SELECT p.id, p.filename FROM photos p, pipeline_routes i
WHERE i.id = :lineId AND ST_DWithin(p.geom, i.geom, 50.0);
\end{lstlisting}
  \begin{itemize}
    \item \textbf{ST\_DWithin:} Fast spatial distance validation using geography coordinates. Automates linking inspection images to asset IDs.
  \end{itemize}
\end{frame}

% Slide 38: Stage 6: Cookie-based session security
\begin{frame}{Stage 6: Cookie Session Guards}
  \begin{itemize}
    \item \textbf{The Risk:} Storing JWTs in LocalStorage exposes tokens to Cross-Site Scripting (XSS).
    \item \textbf{Production Design:}
      \begin{itemize}
        \item Store access tokens in memory or short-lived secure cookies.
        \item Store refresh tokens in HttpOnly, Secure, SameSite cookies.
        \item Rotate refresh tokens on each session update to block token theft.
      \end{itemize}
  \end{itemize}
\end{frame}
